66uint8_t stringToWords( String msg, String words[MAX_WORDS_IN_PROMPT] )
67{
68 uint8_t wordCount = 0; // Number of words in string msg
69 uint8_t idx = 0; // Current char of string msg

75 for( idx; idx < msg.length(); idx++ ) // Looping each character in msg string
